package main

import (
	"fmt"
	server "net/http"
)

func main() {
	fmt.Println("Hello, Go Standard Library")

	rep, err := server.Get("https://jsonplaceholder.typicode.com/posts/1")
	if err != nil {
		fmt.Println("Error: ", err)
		return
	}
	defer rep.Body.Close()

	fmt.Println("HTTP Response Status: ", rep.Status)
}
